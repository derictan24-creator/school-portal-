// 🎓 GRADE FUNCTION
function grade(score){
  if(score>=80) return "A";
  if(score>=70) return "B";
  if(score>=60) return "C";
  if(score>=50) return "D";
  return "F";
}

// 🎓 STUDENT VIEW
window.getResults = async function(){
  let data = await getDocs(collection(db,"results"));

  let total = 0;
  let count = 0;
  let text = "<h4>Results</h4>";

  data.forEach(doc=>{
    let d = doc.data();

    if(d.code === checkCode.value){
      total += d.score;
      count++;

      text += d.subject + ": " + d.score + " ("+grade(d.score)+")<br>";
    }
  });

  if(count === 0){
    output.innerHTML = "No result found";
    return;
  }

  let avg = (total/count).toFixed(2);

  text += "<hr>Total: "+total;
  text += "<br>Average: "+avg;

  output.innerHTML = text;
}
